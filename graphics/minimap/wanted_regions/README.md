## Example of using:
```lua

local wanted_region_is_activated = false
local wanted_region_hash = 0
Citizen.CreateThread(function()
    while true do
        Citizen.Wait(0)
        if IsDisabledControlJustPressed(0, 0x17BEC168) then  -- pressed E
        	if not wanted_region_is_activated then
        		-- apply wanted region blip
        		wanted_region_hash = 0x2D1A7AF2 -- REGION_BLU_SISIKA
        		Citizen.InvokeNative(0x563FCB6620523917, wanted_region_hash, GetHashKey('BLIP_STYLE_WANTED_REGION'));
        		wanted_region_is_activated = true
        	else
        		-- remove wanted region blip
        		Citizen.InvokeNative(0x6786D7AFAC3162B3, wanted_region_hash);
        		wanted_region_is_activated = false
        	end
        end
    end
end)
```
